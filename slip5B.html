# Define the base class Student
class Student:
    def __init__(self, roll_no, name, age, gender):
        self.roll_no = roll_no
        self.name = name
        self.age = age
        self.gender = gender

# Define the subclass Test inheriting from Student
class Test(Student):
    def __init__(self, roll_no, name, age, gender, marks):
        # Initialize parent class attributes
        super().__init__(roll_no, name, age, gender)
        self.marks = marks  # list or tuple of 3 subject marks

    def total_marks(self):
        return sum(self.marks)

    def display(self):
        print("Roll No:", self.roll_no)
        print("Name:", self.name)
        print("Age:", self.age)
        print("Gender:", self.gender)
        print("Marks:", self.marks)
        print("Total Marks:", self.total_marks())
        print("-" * 30)

// A) Write a java program to display alternate character from a given string.
import java.util.Scanner;

public class AlternateCharacters {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        System.out.print("Enter a string: ");
        String str = sc.nextLine();

        System.out.println("Alternate characters are: ");
        for (int i = 0; i < str.length(); i += 2) {
            System.out.print(str.charAt(i));
        }

        sc.close();
    }
}

Example Output:

yaml
Enter a string: Mahabaleshwar
Alternate characters are:
Mhbeshw

Explanation:

  • The loop runs from i = 0 to str.length() - 1.

  • It increases by 2 each time (i += 2) so only every alternate character is printed.

  • str.charAt(i) gives the character at position i.
